    // Boost duration
    setTimeout(() => {
        this.boost = 0;
        console.log("Boost cooling down...");

const dashboard = {
// State
metrics: {
speed: 0,
rpm: 0,
gear: 1,
fuel: 100,
temperature: 180,
boost: 0,
miles: 0
},

```
// Initialize
init() {
    this.createDisplay();
    this.attachControls();
    this.startEngine();
},

// Create dashboard display
createDisplay() {
    document.body.innerHTML = `
        <div id="dashboard">
            <h2>üèéÔ∏è Sports Car Dashboard</h2>
            <div id="gauges">
                <div>Speed: <span id="speed">0</span> mph</div>
                <div>RPM: <span id="rpm">0</span></div>
                <div>Gear: <span id="gear">1</span></div>
                <div>Fuel: <span id="fuel">100</span>%</div>
            </div>
            <button onclick="dashboard.accelerate()">üèÅ Accelerate</button>
            <button onclick="dashboard.brake()">üõë Brake</button>
            <button onclick="dashboard.nitrous()">üí® Nitrous</button>
        </div>
    `;
},

// Control methods
accelerate() {
    if (this.metrics.fuel > 0) {
        this.metrics.speed += 5;
        this.metrics.rpm += 500;
        this.metrics.fuel -= 0.5;
        
        // Auto shift
        if (this.metrics.rpm > 7000) {
            this.shiftUp();
        }
        
        this.update();
    }
},

brake() {
    this.metrics.speed = Math.max(0, this.metrics.speed - 10);
    this.metrics.rpm = Math.max(1000, this.metrics.rpm - 800);
    this.update();
},

nitrous() {
    if (this.metrics.boost > 0) {
        this.metrics.speed += 20;
        this.metrics.boost -= 10;
        console.log("üí® NITROUS BOOST!");
        this.update();
    }
},

shiftUp() {
    if (this.metrics.gear < 6) {
        this.metrics.gear++;
        this.metrics.rpm *= 0.7;
        console.log(`Shifted to gear ${this.metrics.gear}`);
    }
},

update() {
    // Update display
    Object.keys(this.metrics).forEach(key => {
        const element = document.getElementById(key);
        if (element) {
            element.textContent = Math.round(this.metrics[key]);
        }
    });
    
    // Check warnings
    if (this.metrics.fuel < 20) {
        console.warn("‚ö†Ô∏è Low fuel!");
    }
    
    if (this.metrics.rpm > 7500) {
        console.warn("‚ö†Ô∏è Over-revving!");
    }
},

startEngine() {
    console.log("üîë Engine started!");
    this.metrics.rpm = 1000;
    
    // Simulate engine running
    setInterval(() => {
        // Idle consumption
        this.metrics.fuel = Math.max(0, this.metrics.fuel - 0.1);
        
        // Speed affects distance
        this.metrics.miles += this.metrics.speed / 3600;
        
        this.update();
    }, 1000);
}
```

};

// Start when ready
dashboard.init();
